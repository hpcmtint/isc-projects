FROM ubuntu:18.04
WORKDIR /agent
# Install essentials.
RUN apt-get update && apt-get install -y --no-install-recommends sudo curl ca-certificates gnupg apt-transport-https supervisor prometheus-node-exporter net-tools iputils-ping mysql-client
# Install Kea.
RUN curl -1sLf 'https://dl.cloudsmith.io/public/isc/kea-1-8/cfg/setup/bash.deb.sh' | bash
RUN apt-get update && apt-get install -y --no-install-recommends \
        isc-kea-dhcp4-server=1.8.0 \
        isc-kea-ctrl-agent=1.8.0 \
        isc-kea-admin=1.8.0 \
        isc-kea-common=1.8.0 \
        && mkdir -p /var/run/kea/
# Tell CA to listen on all interfaces.
RUN perl -pi -e 's/127\.0\.0\.1/0\.0\.0\.0/g' /etc/kea/kea-ctrl-agent.conf
# Install Stork agent.
COPY build-root/root/usr/bin/stork-agent /agent/
RUN mkdir -p /var/lib/stork-agent/certs /var/lib/stork-agent/tokens
# Copy configuration files.
COPY docker/supervisor-agent-kea.conf /etc/supervisor.conf
COPY docker/kea-dhcp4.conf /etc/kea/
COPY docker/agent-kea-ctrl-agent.conf /etc/kea/kea-ctrl-agent.conf
COPY docker/agent-kea-mysql-init.sh /agent
# Initialize the backends and start the supervisor.
CMD sleep 15 && /agent/agent-kea-mysql-init.sh && supervisord -c /etc/supervisor.conf
